Git example:
git init //to initialize local git repository
echo "newfile contents" > file.txt // to create new file
git status // you can see this file is not tracked
git add . // add files to storage area and so now it is tracked
git commit -m "first commit message"

// connect existing repo to this repo
git remote add origin https://github.com/gokulmvss/Data-Driven-Packet-Loss-Prediction.git
git branch -M main
git push -u origin main

//create new repository
echo "# Data-Driven-Packet-Loss-Prediction" >> README.md
git init
git add README.md
git commit -m "first commit"
git branch -M main
git remote add origin https://github.com/gokulmvss/Data-Driven-Packet-Loss-Prediction.git
git push -u origin main

// to connect github to local git
git config --global user.name "Your Name"
git config --global user.email ""
